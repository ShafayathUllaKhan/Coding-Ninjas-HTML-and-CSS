selectors

1. type selectors
2. class selectors
3. ID selectors
4. universal selectors
5. attribute selectors

4. universal selectors
*{
    styled all the document
}

5. attribute selectors

a[title] {
    only a tag which has title attribute will be styled
}

a[title = "#act"] {
    only a tag which has title attribute will be styled
}

a[title *= "act"] {
    * skip / or # and just check the string
}

a[title ^= "act"] {
    ^ this is like prefix skip # or / from start
}

a[title $= "act"] {
    $ this is like prefix skip # or / from end
}

a[title ^="https"][title $= "codingninjas"]{
    styles https with beginning and ends with codingninjas
}
